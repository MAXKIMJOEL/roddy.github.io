CSS 样式 接下来，我们需要一些 CSS 样式来设计卡片和轮播效果。 Copy /* styles.css */ body { font-family: Arial, sans-serif; display: flex; flex-direction: column; align-items: center; margin: 0; padding: 0; } header { background-color: #333; color: white; padding: 1em; width: 100%; text-align: center; } .carousel { position: relative; width: 80%; max-width: 600px; margin: 20px auto; } .carousel-inner { display: flex; transition: transform 0.5s ease-in-out; } .carousel-item { min-width: 100%; box-sizing: border-box; padding: 10px; } .carousel-item img { width: 100%; border-radius: 10px; } .carousel-caption { text-align: center; margin-top: 10px; } .carousel-control { position: absolute; top: 50%; transform: translateY(-50%); background-color: rgba(0, 0, 0, 0.5); color: white; border: none; padding: 10px; cursor: pointer; } .carousel-control.prev { left: 10px; } .carousel-control.next { right: 10px; } JavaScript 功能 最后，我们需要一些 JavaScript 来实现轮播的功能。 Copy // scripts.js let currentIndex = 0; function changeSlide(direction) { const items = document.querySelectorAll('.carousel-item'); currentIndex = (currentIndex + direction + items.length) % items.length; document.querySelector('.carousel-inner').style.transform = `translateX(-${currentIndex * 100}%)`; }
